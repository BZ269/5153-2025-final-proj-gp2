SELECT COUNT(*) AS entry_count FROM patient_incidents;
SELECT COUNT(patient_id) AS num_cases FROM Inpatient_Cases WHERE treatment_plan IS NULL OR treatment_plan = '';
SELECT hospital_id, AVG(bill_amount) AS avg_billing_amount FROM Outpatient_Cases GROUP BY hospital_id;
SELECT ROUND((SUM(CASE WHEN outcomes = 'Successful' THEN 1 ELSE 0 END) * 100.0 / COUNT(*)), 2) AS success_percentage FROM surgery_metrics;
SELECT pwt.hospital_id, AVG(JULIANDAY(pwt.consultation_end_time) - JULIANDAY(pwt.consultation_start_time)) * 1440 AS avg_consultation_time_minutes FROM patient_wait_times pwt GROUP BY pwt.hospital_id;
SELECT COUNT(DISTINCT ic.patient_id) AS number_of_inpatient_cases FROM Inpatient_Cases ic JOIN surgery_metrics sm ON ic.patient_id = sm.patient_id WHERE ic.follow_up_date IS NOT NULL;
SELECT (COUNT(DISTINCT sm.patient_id) * 100.0 / COUNT(DISTINCT pi.patient_id)) AS surgery_percentage FROM patient_incidents pi LEFT JOIN surgery_metrics sm ON pi.patient_id = sm.patient_id WHERE pi.age > 60;
SELECT department, AVG(duration) AS avg_duration FROM surgery_metrics GROUP BY department ORDER BY avg_duration DESC LIMIT 1;
SELECT AVG(julianday(consultation_start_time) - julianday(registration_time)) AS avg_wait_time FROM patient_wait_times;
SELECT ic.hospital_id, ic.department, COUNT(*) FROM Inpatient_Cases ic JOIN surgery_metrics sm ON ic.patient_id = sm.patient_id JOIN patient_incidents pi ON ic.patient_id = pi.patient_id WHERE pi.age > 60 AND sm.readmission_within_30_days = TRUE GROUP BY ic.hospital_id, ic.department ORDER BY COUNT(*) DESC;
